#include <stdio.h>
int main(void)
{   
    char arr[100];    //배열 받기
    gets(arr);     //띄어쓰기가 포함된 문장을 입력받기

  
    for(int i = 97; i <=103; i++)  // 알파벳 그대로는 받아서 for문 쓸 수 없음=>아스키 코드로 변환
    {
        int num = 0;
        for (int j = 0; arr[j] != '\0'; j++)
        {
            if (arr[j] == i){
                num++;
            }
        }
        printf("%c:%d\n",i,num);       
    }    
    return 0;
}
